Exploring storms and month

ggplot() +
  geom_sf(data = world_map) +
  coord_sf(xlim = c(-110, 0), ylim = c(10, 60)) +
  geom_path(data = storms, 
             aes(x = long, y = lat, group = name),
             alpha = 0.5) +
  facet_wrap(~ month)
storms = storms |> 
  mutate(id = paste0(name, "-", year))

ggplot() +
  geom_sf(data = world_map) +

  geom_path(data = storms, 
             aes(x = long, y = lat, group = id),
             alpha = 0.5, color = "blue") +
  facet_wrap(~ month)

Adding storms in 1975

storms75 |> 
  leaflet() |> 
  addTiles() |> 
  addCircles(lng = ~long,
             lat = ~lat,
             label = ~name, 
             radius = ~1000*wind, 
             stroke = FALSE, 
             fillOpacity = 0.8)

Color Palette

# color palette based on names of storms
# (auxiliary function)
mypal = colorFactor(
  palette = "Dark2",
  domain = storms75$name
)
storms75 |> 
  leaflet() |> 
  addTiles() |> 
  addCircles(lng = ~long,
             lat = ~lat,
             label = ~name,
             color = ~mypal(name),
             radius = ~1000*wind, 
             stroke = FALSE, 
             fillOpacity = 0.8)
storms75 |> 
  leaflet() |> 
  addTiles() |> 
  addCircles(lng = ~long,
             lat = ~lat,
             label = ~name,
             color = ~mypal(name),
             radius = ~1000*wind, 
             stroke = FALSE, 
             fillOpacity = 0.8) |> 
  addLegend(position = "bottomleft", 
            pal = mypal, 
            values = ~name)

Storms in 2010

storms2010 = storms |> filter(year == 2010)
storms2010 |> 
  leaflet() |> 
  addTiles() |> 
  addCircles(lng = ~long,
             lat = ~lat,
             label = ~name)
# another way to create colors
storms2010_colors = storms2010 |> 
  distinct(name) |> 
  mutate(color = rainbow(n()))
storms2010 |> 
  inner_join(storms2010_colors, by = "name") |> 
  leaflet() |> 
  addTiles() |> 
  addCircles(lng = ~long,
             lat = ~lat,
             label = ~paste0(name, "-", month),
             color = ~color,
             radius = ~1000*wind)
